P() - program
<программа> → program { <описания> <операторы> }
D() - descriptions
<описания> → { <описание>; }
D1() - single description
<описание> → <тип> <идентификатор> | <идентификатор> = <константа> {, <идентификатор> |, <идентификатор> = <константа> }

<тип> → int | string | boolean

<переменная> → <идентификатор> | <идентификатор> = <константа>

<константа> → <целочисленная> | <строковая> | <логическая>
<целочисленная> → [<знак>] <цифра> { <цифра> }
<знак> → + | −
<строковая> → " { <литера> } "
<логическая> → true | false

B() - body
<операторы> → { <оператор>; }
S() - statement
<оператор> → if (<выражение>) <оператор> else <оператор> |
	if (<выражение>) <оператор> |
	while (<выражение>) <оператор>|
	read (<идентификатор>); |
	write (<выражение> { , <выражение> } ); |
	<составной оператор> |
	<оператор-выражение> |
	break;
<составной оператор> → { <операторы> }

<оператор-выражение> → <выражение>;
E()	- expression
<выражение> → <r-выражение> | <идентификатор> = <r-выражение> |
E1()
<r-выражение> → <лог-слагаемое> | <r-выражение> or <лог-слагаемое>
E2()
<лог-слагаемое> → <лог-множитель> | <лог-слагаемое> and <лог-множитель>
E3()
<лог-слагаемое> → <сравниваемое> | <лог-слагаемое> > <сравниваемое> |
<лог-слагаемое> < <сравниваемое>
<лог-слагаемое> >= <сравниваемое>
<лог-слагаемое> <= <сравниваемое>
<лог-слагаемое> == <сравниваемое>
<лог-слагаемое> != <сравниваемое>
E4()
<сравниваемое> → <слагаемое> | -<слагаемое> | <сравниваемое> + <слагаемое> | <сравниваемое> - <слагаемое>
E5()
<слагаемое> → <множитель> | <слагаемое> * <множитель> | <слагаемое> / <множитель>
E6()
<множитель> → not <отрицаемое> | <отрицаемое>
E7()
<отрицаемое> → <идентификатор> | <константа> | (<выражение>)
	

	
<идентификатор> → <литера>{<цифра> | <литера> | _}




	not (логическое отрицание)
* / (умножение и деление)
+ − (сложение и вычитание)
< > <= >= == != (операции отношения)
and (логическое умножение)
or (логическое сложение)
= (присваивание)
 

унарные минус и плюс,

«Ленивые» вычисления логических выражений (слева направо; до тех пор,
пока не станет известно значение выражения).
